/* Unity tests for firmware/src/app/rpm_dr.c (dead-reckoning fallback).
 *
 * Runs under both pio test -e native_test (full Unity runtime) and the

#include <math.h>
#include "rpm_dr.h"

/* Helper: build a valid SpectralResult */
static biba_rpm_spectral_result_t make_valid(float freq_hz, float target_hz)
{
    biba_rpm_spectral_result_t r = {0};
    r.valid    = true;
    r.freq_hz  = freq_hz;

    r.freq_hz = 0.0f;
    r.invalid_reason = reason;
    return r;
}

**6 test functions:**

`test_reset_zeroes_state`: Set state.ratio_ema=9.9f, state.streak=7. Call
`biba_rpm_dr_reset(&s)`. Assert `s.ratio_ema == 0.0f` (within 1e-6) and `s.streak == 0`.

`test_cold_start_returns_zero`: Fresh state (reset). Call `biba_rpm_dr_update(&s, invalid(PEAK_LOW), 200.0f, &reason)`. Assert return == 0.0f (within 1e-6) and reason == BIBA_RPM_SPECTRAL_INVALID_PEAK_LOW (not EXTRAPOLATED).

`test_streak_valid_returns_spec_hz`: Fresh state, warm with 2 valid calls at ratio=0.8 (freq=160, target=200). Third call valid, freq=180. Assert return ≈ 180.0f (within 1.0). Assert reason == BIBA_RPM_SPECTRAL_INVALID_NONE. Assert s.streak == 0.

`test_ratio_ema_convergence`: Fresh state. Feed 10 valid calls: freq_hz = 0.9 * target = 180.0f, target = 200.0f. After 10 calls assert ratio_ema >= 0.5f. (EMA of constant 0.9, starting from 0, converges: after 10 steps with α=0.2 → ~0.89*(1-(0.8^10)) ≈ 0.88; floor check of 0.5 is conservative and always passes.)

`test_streak_expiry_returns_zero`: Warm state: 5 valid calls (ratio=0.8, freq=160, target=200) → ratio_ema ≈ 0.54. Then feed BIBA_RPM_DR_MAX_STREAK+2 consecutive invalid calls at target=200Hz. Capture all return values. Assert that the call at index BIBA_RPM_DR_MAX_STREAK (0-indexed, the "last allowed") returns > 0.0f. Assert that the call at index BIBA_RPM_DR_MAX_STREAK+1 returns 0.0f.

`test_ratio_clamp_lo`: Fresh state. Feed one valid call: freq_hz=40.0f (ratio=0.2 below LO=0.5), target=200.0f. Assert ratio_ema == BIBA_RPM_DR_RATIO_LO * BIBA_RPM_DR_ALPHA (within 1e-4). (Because ratio is clamped to 0.50 before EMA: ratio_ema = 0.2*0.50 + 0.8*0.0 = 0.10.)

`test_ratio_clamp_hi`: Fresh state. Feed one valid call: freq_hz=600.0f (ratio=3.0 above HI=1.30), target=200.0f. Assert ratio_ema <= BIBA_RPM_DR_RATIO_HI (within 1e-4).
