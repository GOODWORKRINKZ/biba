  Script header:
    #!/usr/bin/env python3
    """Battery sag cross-talk analysis — DC_L vs DC_R Pearson-r from softhold capture."""

  Imports: pathlib.Path, sys, numpy as np, pandas as pd, scipy.stats.pearsonr,
           matplotlib (use("Agg")), matplotlib.pyplot as plt.

  Constants:
    BASE_DIR  = Path(__file__).resolve().parent / "artifacts" / "is-sweepraw"
    CSV_STEM  = "sweepraw_TRAP_amp50_per6000_n60_20260526-135642_softhold"
    OUT_PNG   = Path(__file__).resolve().parent / "artifacts" / "batsag_scatter.png"
    R_TRIGGER = 0.3   # threshold for recommending controlled capture (per D-C2)

  Data loading (same pattern as plan 11-01):
    Read CSV_STEM + "_left.csv" and "_right.csv" with pd.read_csv.
    Compute dc_l = df_l.groupby("win_idx")["adc_raw"].mean() (Series indexed by win_idx).
    Compute dc_r = df_r.groupby("win_idx")["adc_raw"].mean().
    Align on common win_idx via inner join: merged = dc_l.to_frame("dc_l").join(dc_r.to_frame("dc_r"), how="inner").reset_index().

  Statistics:
    r, p = scipy.stats.pearsonr(merged["dc_l"].values, merged["dc_r"].values)
    Print:
      f"DC_L mean={merged['dc_l'].mean():.1f}  DC_R mean={merged['dc_r'].mean():.1f}"
      f"Pearson-r = {r:.3f}   p = {p:.3e}   n = {len(merged)}"
    Print the CONFIRMED line:
      f"CONFIRMED: battery sag correlation |r|={abs(r):.3f}"

  Controlled capture recommendation (per D-C2):
    if abs(r) > R_TRIGGER:
        print("")
        print("Next step: capture with one motor held (stall), other free.")
        print("  Command: python3 scripts/is_poc_sweepraw.py --port /dev/ttyACM0 \\")
        print("           --shape TRAP --amp 50 --period 6000 --n-windows 60 \\")
        print("           --motor both --tag stall_L_free_R --no-analyze")
        print("  Goal: compute k_sag = mean(ΔDC_free) / DC_free_base (per D-C2).")

  Figure (single subplot):
    figsize=(7, 6)
    Scatter: x=merged["dc_r"], y=merged["dc_l"], c=merged["win_idx"], cmap="viridis", alpha=0.7, s=40
    Add colorbar labelled "win_idx".
    Overplot best-fit line: np.polyfit → np.polyval, color="red", lw=1.5, label=f"fit (r={r:.3f})".
    x-label: "DC_R (ADC counts)", y-label: "DC_L (ADC counts)"
    Title: f"Battery Sag Cross-talk — Pearson r={r:.3f}  p={p:.2e}"
    legend(loc="upper left")
    Save to OUT_PNG, bbox_inches="tight", dpi=150.
    Print: f"saved: {OUT_PNG}"
  </action>
